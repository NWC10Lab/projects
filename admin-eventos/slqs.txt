sql para saber los eventos de un usuario por su id y activo

select e.nombre , e.id 
from eventos as e
inner join usuarios_eventos ue on ue.id_evento=e.id
inner join usuarios u on u.id=ue.id_usuario
where u.id=$id and e.fecha_baja is NULL



sql para saber los eventos de un usuario por su id y INACTIVO

select e.nombre , e.id 
from eventos as e
inner join usuarios_eventos ue on ue.id_evento=e.id
inner join usuarios u on u.id=ue.id_usuario
where u.id=$id and e.fecha_baja is not NULL
